294:       incl, excl, replace = [], [], false
295: 
296:       File.read('inclexcl').split("\n").each { |pattern|
297:         pattern.sub!(/\A\s*/, '')

302:         prefix = $1  # save capture
303:         next if prefix == '#'

307:           warn "[ERROR] Illegal pattern - #{pattern}"
